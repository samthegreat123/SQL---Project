CREATE TABLE CUSTOMERS(
customer_id NUMBER PRIMARY KEY NOT NULL,
first_name VARCHAR(20) NOT NULL,
last_name VARCHAR(20) NOT NULL,
date_of_birth DATE NOT NULL,
gender CHAR(1) CHECK (gender IN ('M', 'F', 'O')),
marital_status VARCHAR2(10) DEFAULT 'SINGLE',
occupation VARCHAR2(50),
annual_income NUMBER(10,2) CHECK (annual_income >=0),
registration_date DATE DEFAULT SYSDATE
);

CREATE TABLE ADDRESSES(
address_Id NUMBER PRIMARY KEY NOT NULL,
customer_id NUMBER NOT NULL, FOREIGN KEY(customer_id) REFERENCES CUSTOMERS(customer_id),
address_line1 VARCHAR(100) NOT NULL,
address_line2 VARCHAR(100),
city VARCHAR2(50) NOT NULL,
state VARCHAR2(50) NOT NULL,
postal_code VARCHAR2(20) NOT NULL,
country VARCHAR2(50) DEFAULT 'LIBERIA' NOT NULL,
address_type VARCHAR2(20) CHECK(address_type IN ('HOME','WORK','MAILING','PERMANENT')) NOT NULL
);

CREATE TABLE CONTACT(
contact_id NUMBER PRIMARY KEY NOT NULL,
customer_id NUMBER NOT NULL, FOREIGN KEY(customer_id) REFERENCES CUSTOMERS(customer_id),
contact_type VARCHAR2(20) CHECK(contact_type IN('PHONE','EMAIL','MOBILE','FAX')) NOT NULL,
contact_value VARCHAR2(100) NOT NULL,
is_primary VARCHAR2(3) CHECK(is_primary IN('YES','NO')) DEFAULT 'NO',
verified_status VARCHAR2(10) CHECK(verified_status IN('VERIFIED', 'PENDING', 'FAILED')) DEFAULT 'PENDING'
);

CREATE TABLE IDENTIFICATION_DOCS(
doc_id NUMBER PRIMARY KEY NOT NULL,
customer_id NUMBER NOT NULL, FOREIGN KEY(customer_id) REFERENCES CUSTOMERS(customer_id),
doc_type VARCHAR2(20) CHECK (doc_type IN ('PASSPORT', 'DRIVING_LICENSE', 'NATIONAL_ID', 'VOTER_ID', 'PAN_CARD', 'SSN')) NOT NULL,
doc_number VARCHAR2(50) NOT NULL,
issue_date DATE NOT NULL,
expiry_date DATE NOT NULL,
issuing_authority VARCHAR2(100) NOT NULL,
verified_status VARCHAR2(10) DEFAULT 'PENDING' CHECK (verified_status IN ('VERIFIED', 'PENDING', 'REJECTED'))
);

CREATE TABLE ACCOUNT_TYPES(
account_type_id NUMBER PRIMARY KEY NOT NULL,
type_name VARCHAR2(30) UNIQUE NOT NULL,
description VARCHAR2(50),
minimum_bal NUMBER DEFAULT 0 CHECK(minimum_bal >=0),
interest_rate NUMBER DEFAULT 0 CHECK(interest_rate>=0),
monthly_fee NUMBER DEFAULT 0 CHECK(monthly_fee>=0),
transaction_limit_per_day NUMBER DEFAULT 0 CHECK(transaction_limit_per_day>=0),
withdrawl_limit_per_day NUMBER DEFAULT 0 CHECK(withdrawl_limit_per_day>=0)
);

CREATE TABLE ACCOUNT_STATUS(
account_status_id NUMBER PRIMARY KEY NOT NULL,
status_name VARCHAR2(30) UNIQUE NOT NULL,
description VARCHAR2(40)
);

CREATE TABLE ACCOUNTS(
account_id NUMBER PRIMARY KEY NOT NULL,
customer_id NUMBER NOT NULL,FOREIGN KEY (customer_id) REFERENCES CUSTOMERS(customer_id),
account_number NUMBER UNIQUE NOT NULL,
account_type_id NUMBER NOT NULL, FOREIGN KEY(account_type_id) REFERENCES ACCOUNT_TYPES(account_type_id),
account_status_id NUMBER NOT NULL, FOREIGN KEY(account_status_id) REFERENCES ACCOUNT_STATUS(account_status_id),
current_bal NUMBER DEFAULT 0 CHECK(current_bal>=0),
available_bal NUMBER DEFAULT 0 CHECK(available_bal>=0),
currency VARCHAR2(10) DEFAULT'LD' NOT NULL,
opening_date DATE DEFAULT SYSDATE NOT NULL
);

CREATE TABLE TRANSACTION_TYPES(
transaction_types_id NUMBER PRIMARY KEY NOT NULL,
type_name VARCHAR2(20) UNIQUE NOT NULL,
description VARCHAR2(100)
);

CREATE TABLE TRANSACTION_STATUS(
transaction_status_id NUMBER PRIMARY KEY NOT NULL,
status_name VARCHAR2(50) UNIQUE NOT NULL,
description VARCHAR2(100)
);

CREATE TABLE TRANSACTIONS(
transaction_id NUMBER PRIMARY KEY NOT NULL,
account_id NUMBER NOT NULL, FOREIGN KEY(account_id) REFERENCES ACCOUNTS(account_id),
transaction_type_id NUMBER NOT NULL, FOREIGN KEY(transaction_type_id) REFERENCES TRANSACTION_TYPES(transaction_types_id),
transaction_status_id NUMBER NOT NULL, FOREIGN KEY(transaction_status_id) REFERENCES TRANSACTION_STATUS(transaction_status_id),
amount NUMBER NOT NULL CHECK(amount>0),
transaction_date DATE DEFAULT SYSDATE NOT NULL,
description VARCHAR2(100),
reference_number VARCHAR2(50) UNIQUE,
from_account_id NUMBER, FOREIGN KEY(from_account_id) REFERENCES ACCOUNTS(account_id),
to_account_id NUMBER, FOREIGN KEY(to_account_id) REFERENCES ACCOUNTS(account_id),
balance_after_transaction NUMBER NOT NULL
);

CREATE TABLE CARD_TYPES(
card_type_id NUMBER PRIMARY KEY NOT NULL,
type_name VARCHAR2(20) UNIQUE NOT NULL,
description VARCHAR2(100)
);

CREATE TABLE CARDS(
card_id NUMBER PRIMARY KEY NOT NULL,
account_id NUMBER NOT NULL, FOREIGN KEY(account_id) REFERENCES ACCOUNTS(account_id),
card_type_id NUMBER NOT NULL, FOREIGN KEY(card_type_id) REFERENCES CARD_TYPES(card_type_id),
card_number VARCHAR2(20) UNIQUE NOT NULL,
cardholder_name VARCHAR2(100) NOT NULL,
expiry_date DATE NOT NULL,
cvv VARCHAR2(10) NOT NULL,
issue_date DATE DEFAULT SYSDATE NOT NULL,
daily_limit NUMBER DEFAULT 0 CHECK(daily_limit>=0),
status VARCHAR2(10) NOT NULL CHECK(status IN('ACTIVE','BLOCKED','EXPIRED','LOST','STOLEN'))
);

CREATE TABLE CHEQUES(
cheque_id NUMBER PRIMARY KEY NOT NULL,
account_id NUMBER NOT NULL, FOREIGN KEY(account_id) REFERENCES ACCOUNTS(account_id),
cheque_number VARCHAR2(20) NOT NULL,
issue_date DATE DEFAULT SYSDATE NOT NULL,
amount NUMBER NOT NULL CHECK(amount>0),
payee_name VARCHAR2(100) NOT NULL,
status VARCHAR2(20) NOT NULL CHECK(status IN('PENDING','CLEARED','BOUNCED','CANCELLED','STOPPED')),
cleared_date DATE,
remarks VARCHAR2(200)
);

CREATE TABLE LOANS(
loan_id NUMBER PRIMARY KEY NOT NULL,
account_id NUMBER NOT NULL, FOREIGN KEY(account_id) REFERENCES ACCOUNTS(account_id),
loan_type VARCHAR2(20) NOT NULL CHECK(loan_type IN('PERSONAL','HOME','AUTO','EDUCATION','BUSINESS')),
principal_amount NUMBER NOT NULL CHECK(principal_amount>0),
interest_rate NUMBER NOT NULL CHECK(interest_rate>=0),
loan_term_months NUMBER NOT NULL CHECK(loan_term_months>0),
emi_amount NUMBER NOT NULL CHECK(emi_amount>0),
total_payable NUMBER NOT NULL CHECK(total_payable>0),
amount_paid NUMBER DEFAULT 0 CHECK(amount_paid>=0),
outstanding_balance NUMBER NOT NULL CHECK(outstanding_balance>=0),
sanction_date DATE DEFAULT SYSDATE,
loan_status VARCHAR2(20) NOT NULL CHECK(loan_status IN('ACTIVE','CLOSED','DEFAULT','PENDING_APPROVAL'))
);

CREATE TABLE LOAN_PAYMENTS(
payment_id NUMBER PRIMARY KEY NOT NULL,
loan_id NUMBER NOT NULL, FOREIGN KEY(loan_id) REFERENCES LOANS(loan_id),
payment_date DATE DEFAULT SYSDATE NOT NULL,
amount_paid NUMBER NOT NULL CHECK(amount_paid >0),
payment_status VARCHAR2(20) NOT NULL CHECK(payment_status IN ('SUCCESS','FAILED','PENDING')),
payment_method VARCHAR2(15) NOT NULL CHECK(payment_method IN('AUTO_DEBIT','MANUAL','CASH','CHEQUE','TRANSFER')),
remarks VARCHAR2(200)
);

CREATE TABLE LOAN_EMI_SCHEDULE(
schedule_id NUMBER PRIMARY KEY NOT NULL,
loan_id NUMBER NOT NULL, FOREIGN KEY(loan_id) REFERENCES LOANS(loan_id),
installment_number NUMBER CHECK(installment_number>0),
due_date DATE NOT NULL,
emi_amount NUMBER NOT NULL CHECK(emi_amount>0),
principal_component NUMBER NOT NULL CHECK(principal_component>0),
interest_component NUMBER NOT NULL CHECK(interest_component>0),
outstanding_balance_after NUMBER NOT NULL CHECK(outstanding_balance_after>=0),
payment_status VARCHAR2(20) DEFAULT 'UNPAID' CHECK(payment_status IN ('PAID','UNPAID','PARTIAL','OVERDUE'))
);


CREATE TABLE BRANCHES(
branch_id NUMBER PRIMARY KEY NOT NULL,
branch_name VARCHAR2(40) UNIQUE NOT NULL,
branch_code VARCHAR2(20) UNIQUE NOT NULL,
address_line1 VARCHAR2(40) NOT NULL,
address_line2 VARCHAR2(20),
city VARCHAR2(20) NOT NULL,
state VARCHAR2(20) NOT NULL,
postal_code VARCHAR2(20) NOT NULL,
country VARCHAR2(30) DEFAULT 'LIBERIA' NOT NULL,
phone VARCHAR2(20) NOT NULL,
email VARCHAR2(40),
manager_id NUMBER,
opening_date DATE DEFAULT SYSDATE NOT NULL,
status VARCHAR2(20) DEFAULT 'ACTIVE' CHECK(status IN('ACTIVE','CLOSED','TEMPORARILY_CLOSED'))
);

CREATE TABLE EMPLOYEES(
employee_id NUMBER PRIMARY KEY NOT NULL,
branch_id NUMBER NOT NULL, FOREIGN KEY(branch_id) REFERENCES BRANCHES(branch_id),
first_name VARCHAR2(20) NOT NULL,
last_name VARCHAR2(20) NOT NULL,
date_of_birth DATE NOT NULL,
gender CHAR(1) CHECK (gender IN('M','F','O')),
email VARCHAR2(20) UNIQUE NOT NULL,
phone VARCHAR2(20) NOT NULL,
hire_date DATE DEFAULT SYSDATE NOT NULL,
job_title VARCHAR2(30) NOT NULL CHECK(job_title IN ('TELLER','LOAN_OFFICER','BRANCH_MANAGER','ACCOUNTANT','CUSTOMER_SERVICE','SECURITY')),
salary NUMBER NOT NULL CHECK(salary >=0),
manager_id NUMBER, FOREIGN KEY(manager_id) REFERENCES EMPLOYEES(employee_id),
employment_status VARCHAR2(30) DEFAULT 'ACTIVE' CHECK(employment_status IN ('ACTIVE','RESIGNED','TERMINATED','ON_LEAVE'))
);

CREATE TABLE INTEREST_RATES(
rate_id NUMBER PRIMARY KEY NOT NULL,
product_type VARCHAR2(20) NOT NULL CHECK(product_type IN('SAVINGS','LOAN','FIXED_DEPOSIT','CURRENT')),
interest_rate NUMBER NOT NULL CHECK(interest_rate >=0),
effective_from DATE NOT NULL,
effective_to DATE,
is_active VARCHAR2(3) DEFAULT 'YES' CHECK(is_active IN('YES','NO'))
);

CREATE TABLE FEES(
fee_id NUMBER PRIMARY KEY NOT NULL,
fee_name VARCHAR2(40) UNIQUE NOT NULL,
fee_type VARCHAR2(20) NOT NULL CHECK(fee_type IN('MONTHLY','TRANSACTION','PENALTY','SERVICE','MAINTENANCE')),
amount NUMBER NOT NULL CHECK(amount>=0),
applicable_to VARCHAR2(20) NOT NULL CHECK(applicable_to IN('SAVINGS','CURRENT','LOAN','CREDIT_CARD','ALL')),
effective_from DATE NOT NULL,
effective_to DATE,
is_active VARCHAR2(3) DEFAULT 'YES' CHECK(is_active IN('YES','NO'))
);
